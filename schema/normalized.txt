// I used dbdiagram.io for this

Table Measurements as M {
  id int  [pk]
  ts Timestamp 
  station int
  PM2dot5 double // can't use dots as column names
  PM10 double
  SO2 double
  NO2 double
  CO double
  O3 double
  TEMP double
  PRES double
  DEWP double
  RAIN double
  wd varchar
  WSPM double
}

Table Station as S {
  id int  [pk]
  station varchar 
}


Ref {
  M.station > S.station
}
